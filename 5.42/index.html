　　装饰器：在不修改原函数内容及调用方式上，实现函数的功能扩展

import time
def timmer(func):
def inner(*args,**kwargs):
start=time.time()
ret=func(*args,**kwargs)
end=time.time()
print(end-start)
return ret
return inner
@timmer
def name(a,b):
print('%s+%s等于%s' %(a,b,a+b))
